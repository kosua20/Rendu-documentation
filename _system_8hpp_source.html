59 static void forParallel(size_t low, size_t high, ThreadFunc func) {
60 // Make sure the loop is increasing.
61 if(high < low) {
62 const size_t temp = low;
63 low = high;
64 high = temp;
65 }
66 // Prepare the threads pool.
67 // Always leave one thread free.
68 const size_t availableThreadsCount = std::max(int(std::thread::hardware_concurrency())-1, 1);
69 const size_t count = std::min(size_t(availableThreadsCount), high - low);
70 std::vector<std::thread> threads;
71 threads.reserve(count);
72
73 // Compute the span of each thread.
74 const size_t span = std::max(size_t(1), (high - low) / count);
75 // Helper to execute the function passed on a subset of the total interval.
76 auto launchThread = [&func](size_t a, size_t b) {
77 for(size_t i = a; i < b; ++i) {
78 func(i);
79 }
80 };
81
82 for(size_t tid = 0; tid < count; ++tid) {
83 // For each thread, call the same lambda with different bounds as arguments.
84 const size_t threadLow = tid * span;
85 const size_t threadHigh = tid == (count-1) ? high : ((tid + 1) * span);
86 threads.emplace_back(launchThread, threadLow, threadHigh);
87 }
88 // Wait for all threads to finish.
89 std::for_each(threads.begin(), threads.end(), [](std::thread & x) { x.join(); });
